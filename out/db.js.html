/**
 * Return the from selected genre the question object
 * 
 * @returns {Array}
 */
function getSelectedQuestionDB() {
    if (selectedQuestionDB == 'HTML') {
        return htmlQuestions;
    } if (selectedQuestionDB == 'CSS') {
        return cssQuestions;
    } else {
        return jsQuestions;
    }
}
